# The wheel, repeated to be >192 bits
wheel = '11100101100'
for i in range(0, 18):
    wheel += '11100101100'

# Permutations of wheel
# 'iter' is the offset to test
for iter in range(0, 11):
    password = ''
    for idx, bit in enumerate(list(input)):
        if bit != list(wheel)[idx+iter]:
            password += '1'
        else:
            password += '0'

    print('Password %d: %s' % (iter, password))
